# 자리수보정 KST로만든 STRING
def tof(x) :
    x=x.map(lambda x : str(x).replace('.500000+09:00','') )
    x=x.map(lambda x : str(x).replace('+09:00','') )
    x=x.map(lambda x : datetime.datetime.strptime(x,'%Y-%m-%d %H:%M:%S') )
    return x
